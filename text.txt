#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include <time.h>

/*row swap*/
void swap(double A[4][4], int m, int n)
{
    double t;
    for(int i = 0; i< 4; i++)
    {
        t = A[m][i];
        A[m][i] = A[n][i];
        A[n][i] = t;
    }
}

/*row plus k times*/
void plus(double A[4][4], int m/*be plused*/, int n/*plus to m*/, double k)
{
    for(int i = 0; i<4; i++)
    {
        A[m][i]+=A[n][i]*k;
    }
}
/*have a main element*/
int have(double A[4][4], int row_start, int col_start)
{
    for(int circle = 1; circle<= 3-row_start; circle ++)
    {
        for(int i = row_start; i<4; i++)
        {
            if (A[i][col_start]< A[i+1][col_start])
            {
                swap(A, i, i+1);
            }
        }
    }
    if (A[row_start][col_start]==0) return 0;
    if (A[row_start][col_start]!=0) return 1;
}

int main()
{
    /*create a matrix*/
    srand(time(NULL));
    double A[4][4];
    for(int i = 0; i< 4; i++)

        for(int j = 0; j< 4; j++)

            A[i][j] = rand()%101;
        }

    /*run through rows*/
    int i = 0;

        int bull = have(A,i,j);
        i+=bull;

        {
